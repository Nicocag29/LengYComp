% preamble.tex — paquetes, estilo y macros de notación
% Apunte LyC (Reynolds, Theories of Programming Languages, 2009)

% --- Idioma y codificación ---
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage[spanish]{babel}

% --- Color (antes de tikz/forest para evitar option clash con la opción [table]) ---
\usepackage[table,dvipsnames]{xcolor}

% --- Matemática ---
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{amsthm}
\usepackage{mathtools}
\usepackage{stmaryrd}

% --- Layout y tipografía ---
\usepackage[a4paper,margin=2.5cm]{geometry}
\usepackage{microtype}
\usepackage{booktabs}
\usepackage{array}
\usepackage{tabularx}
\usepackage{enumitem}

% --- Cajas y marcos ---
\usepackage[framemethod=TikZ]{mdframed}

% --- Árboles ---
\usepackage{bussproofs} % árboles de derivación
\usepackage{forest}     % árboles de sintaxis

% --- Teoremas con estilo (sobre amsthm) ---
\usepackage{thmtools}

% --- Encabezados y pies ---
\usepackage{titlesec}
\usepackage{fancyhdr}

% --- Enlaces (cerca del final) ---
\usepackage{hyperref}

% ====================== Paleta ======================
\definecolor{accent}{HTML}{1F3A5F}   % azul pizarra (títulos, enlaces)
\definecolor{defmain}{HTML}{2C5F8A}  \definecolor{defbg}{HTML}{EEF4FA}
\definecolor{thmmain}{HTML}{8A2B2B}  \definecolor{thmbg}{HTML}{FBEFEF}
\definecolor{propmain}{HTML}{1F6B4C} \definecolor{propbg}{HTML}{EBF5EF}
\definecolor{exmain}{HTML}{8A671F}   \definecolor{exbg}{HTML}{FAF5E8}

% ====================== hyperref ======================
\hypersetup{
  plainpages=false,   % evita destinos PDF duplicados (page.1)
  colorlinks=true,
  linkcolor=accent,
  citecolor=accent,
  urlcolor=accent,
  bookmarksnumbered=true,
}

% ====================== Encabezados de sección ======================
\titleformat{\chapter}[display]
  {\sffamily\bfseries\color{accent}}
  {\filright\large\MakeUppercase{\chaptertitlename}\ \thechapter}
  {6pt}{\Huge\filright}
\titlespacing*{\chapter}{0pt}{10pt}{24pt}

\titleformat{\section}
  {\sffamily\bfseries\Large\color{accent}}
  {\thesection}{0.7em}{}
\titleformat{\subsection}
  {\sffamily\bfseries\large\color{accent!88!black}}
  {\thesubsection}{0.6em}{}

% ====================== Encabezado / pie de página ======================
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\headrule}{\hbox to\headwidth{\color{accent!40}\leaders\hrule height \headrulewidth\hfill}}
\fancyhead[L]{\small\sffamily\color{accent}\nouppercase{\leftmark}}
\fancyhead[R]{\small\sffamily\color{accent}\thepage}
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\small\sffamily\color{accent}\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% ====================== Tablas ======================
\renewcommand{\arraystretch}{1.25}

% ====================== Teoremas (cajas mdframed) ======================
\declaretheoremstyle[
  headfont=\sffamily\bfseries\color{defmain},
  notefont=\normalfont\sffamily\color{defmain!85!black},
  bodyfont=\normalfont,
  spaceabove=10pt, spacebelow=10pt,
  mdframed={
    linewidth=0.9pt, linecolor=defmain!55, backgroundcolor=defbg,
    roundcorner=4pt,
    innertopmargin=7pt, innerbottommargin=7pt,
    innerleftmargin=10pt, innerrightmargin=10pt,
  },
]{defstyle}

\declaretheoremstyle[
  headfont=\sffamily\bfseries\color{thmmain},
  notefont=\normalfont\sffamily\color{thmmain!85!black},
  bodyfont=\itshape,
  spaceabove=10pt, spacebelow=10pt,
  mdframed={
    linewidth=0.9pt, linecolor=thmmain!55, backgroundcolor=thmbg,
    roundcorner=4pt,
    innertopmargin=7pt, innerbottommargin=7pt,
    innerleftmargin=10pt, innerrightmargin=10pt,
  },
]{thmstyle}

\declaretheoremstyle[
  headfont=\sffamily\bfseries\color{propmain},
  notefont=\normalfont\sffamily\color{propmain!85!black},
  bodyfont=\itshape,
  spaceabove=10pt, spacebelow=10pt,
  mdframed={
    linewidth=0.9pt, linecolor=propmain!55, backgroundcolor=propbg,
    roundcorner=4pt,

  },

\declaretheoremstyle[
  headfont=\sffamily\bfseries\color{exmain},
  notefont=\normalfont\sffamily\color{exmain!85!black},
  bodyfont=\normalfont,
  spaceabove=10pt, spacebelow=10pt,
  mdframed={

  },

\declaretheorem[style=thmstyle,  name=Teorema,     numberwithin=chapter]{theorem}
\declaretheorem[style=propstyle, name=Proposición, sibling=theorem]{proposition}
\declaretheorem[style=defstyle,  name=Definición,  sibling=theorem]{definition}
\declaretheorem[style=exstyle,   name=Ejemplo,     sibling=theorem]{example}

\newtheorem{ejercicio}{Ejercicio}
